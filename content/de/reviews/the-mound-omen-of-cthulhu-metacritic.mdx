  <Step num="1" title="Ausrüstungsauswahl">

  </Step>
  <Step num="2" title="Dschungel-Navigation">

  <Step num="3" title="Ressourcenmanagement">

  </Step>
  <Step num="4" title="Betreten von The Mound">
    Erreiche die legendären Tore der Unterwelt. Erwarte einen ansteigenden Schwierigkeitsgrad, je tiefer du in den Abgrund vordringst.
  </Step>
</Steps>
